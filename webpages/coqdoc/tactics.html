            | [ H : beq_nat ?X ?Y = _ |- _ ] => match goal with | [H1: context [ if beq_nat ?X ?Y then _ else _ ] |- _ ] => rewrite H in H1; simpl in H1 end end; try assumption ; try reflexivity.
